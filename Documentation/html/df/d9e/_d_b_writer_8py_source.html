672  def _LocationCaseTwo(self, inArticleID: int, inLocation: str, inAdminList: List[Tuple]) -> bool:
673  """
674  Check to see if any of the locations are in any of the administrative areas in adminList. We commit IFF we get
675  a single match for any of the locations.
676  :param inArticleID: ID of the article we're processing
677  :param inLocation: Input location string to check
678  :param inAdminList: list of tuples with administative names and their CC2 codes.
679  :return: True if we wrote the entry to the database
680  """
681 
682  if not inAdminList or not inLocation:
683  return False
684 
685  try:
686  for tempAdmin in inAdminList:
687  # FIXME: Skip for now if it's a country, may want to add it in the future.
688  # if inLocation in dict(inAdminList):
689  # continue
690 
691  # First check if we get a single match for the location vs the country.
692  self._DBCursor.execute("SELECT id FROM locations WHERE name = %s AND cc2 = %s;",
693  (inLocation, tempAdmin[1]))
694  self._DBConnection.commit()
695 
696  if self._DBCursor.rowcount == 1:
697  locationID = self._DBCursor.fetchone()
698  self._InsertArticleLocations(inArticleID, locationID[0])
699  return True
700  elif self._DBCursor.rowcount > 1:
701  # Pick the match with the highest population.
702  self._DBCursor.execute("""
703  SELECT id, population FROM locations WHERE name = %s AND cc2 = %s AND
704  population IS NOT NULL ORDER BY population DESC LIMIT 1;
705  """, (inLocation, tempAdmin[1]))
706  self._DBConnection.commit()
707  locationID = self._DBCursor.fetchone()
708  self._InsertArticleLocations(inArticleID, locationID[0])
709  return True
710 
711  # Second, check for a similarity score but we only use it if we get a single result.
712  self._DBCursor.execute("""
713  SELECT id FROM locations WHERE SIMILARITY(name, %s) > 0.5 AND cc2 = %s;
714  """, (inLocation, tempAdmin[1]))
715  self._DBConnection.commit()
716  if self._DBCursor.rowcount == 1:
717  locationID = self._DBCursor.fetchone()
718  self._InsertArticleLocations(inArticleID, locationID[0])
719  return True
